# Collation of Sidama letters

& d < dh <<< Dh <<< DH
& n < ny <<< Ny <<< NY
& p < ph <<< Ph <<< PH
& s < sh <<< Sh <<< SH
& t < ts <<< Ts <<< TS
& z < zh <<< Zh <<< ZH
